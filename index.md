function slowsort(A, i, j) {
  if (i >= j)
    return null;

  m = (i + j) >> 1;
  A = slowsort(A, i, m);
  A = slowsort(A, m + 1, j);

  if (A[m] > A[j])
    [A[m], A[j]] = [A[j], A[m]];

  return slowsort(A, i, j - 1);
}
